const testimonials = [ // Array storing testimonials
  { name: "Cherise G", photoUrl: "image1.jpg", text: "Great product!" },
  { name: "Rosetta Q", photoUrl: "image2.jpg", text: "Amazing service!" },
  { name: "Constantine V", photoUrl: "image3.jpg", text: "Highly recommended!" }
];

const imgEl = document.querySelector("img"); // Selects image element
const textEl = document.querySelector(".text"); // Selects text container
const usernameEl = document.querySelector(".username"); // Selects name container

let idx = 0; // Keeps track of current testimonial

updateTestimonial(); // Starts the first testimonial display

function updateTestimonial() {
  const { name, photoUrl, text } = testimonials[idx]; // Extracts current testimonial
  imgEl.src = photoUrl; // Updates image
  textEl.innerText = text; // Updates testimonial text
  usernameEl.innerText = name; // Updates name
  
  idx++; // Move to the next testimonial
  if (idx === testimonials.length) idx = 0; // Reset index if at the end

  setTimeout(updateTestimonial, 10000); // Calls function every 10 seconds
}
